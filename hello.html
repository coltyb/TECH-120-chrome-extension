from jinja2 import Template # Define the HTML template template_str = """

Hello Extensions

""" # Compile the template template = Template(template_str) # Define the variables to be used in the template title = "Hello, world!" message = "This text is being printed from a Python program using Jinja2." # Render the template with the variables output_html = template.render(title=title, message=message) # Print the output HTML print(output_html)
